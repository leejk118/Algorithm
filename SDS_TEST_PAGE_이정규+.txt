#include <iostream>
#include <algorithm>
using namespace std;

int main() {

	int T;
	cin >> T;
	for (int i = 0; i < T; i++) {
		int N, S, J, K, page[1000];
		cin >> N >> S >> J >> K;
		for (int j = 0; j < K; j++) {
			cin >> page[j];
		}
		int answer = 0;

		sort(page, page + K);

		int pageIndex = 0;
		int currentPage = S;

		while (currentPage <= N) {
			if (pageIndex > K) break;
			if (currentPage == page[pageIndex]) {
				answer++;
				pageIndex++;
				currentPage += (J + 1);
			}
			else if (currentPage > page[pageIndex]) {
				pageIndex++;
				continue;
			}
			else {
				currentPage += (J + 1);
			}
		}
		cout << "#" << i + 1  << " " << answer << endl;
	}

	return 0;
}
